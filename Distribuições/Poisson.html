library(ggplot2)

# Parmetro da distribuio
lambda <- 3.5

# Suporte de k (nmero de ocorrncias possveis)
k <- 0:15

# Data frame com probabilidades
pois <- data.frame(
  k = k,
  P = dpois(k, lambda)
)

# Grfico com ggplot2
ggplot(pois, aes(x = k, y = P)) +
  geom_segment(aes(xend = k, yend = 0), linewidth = 1.2, color = "#6fa8dc") +
  geom_point(size = 3, color = "#f78181") +
  geom_text(aes(label = round(P, 3)), vjust = -0.6, size = 4) +
  labs(
    title = paste0("Distribuio Poisson( = ", lambda, ")"),
    x = "Nmero de ocorrncias (k)",
    y = "Probabilidade"
  ) +
  theme_minimal(base_size = 14)

# Gerar e visualizar uma amostra Poisson

# --- 1. Definir Parmetros ---
n <- 1000        # tamanho da amostra
lambda0 <- 4     # mdia verdadeira (parmetro  da Poisson)

# --- 2. Gerar Amostra Aleatria ---
# Usamos rpois() para "random poisson"
x <- rpois(n, lambda = lambda0)

# --- 3. Histograma + Probabilidade Terica ---

# Para dados discretos (inteiros),  bom ajustar os 'breaks' (barras)
# do histograma para que cada barra represente um nmero inteiro.
breaks <- seq(-0.5, max(x) + 0.5, by = 1)

# hist() desenha a amostra observada
hist(x,
     prob = TRUE,               # Mostra proporo (densidade) em vez de contagem
     breaks = breaks,           # Alinha as barras aos inteiros
     col = "lightblue",
     main = paste("Distribuio Poisson ( =", lambda0, ")"),
     xlab = "k (Nmero de eventos)",
     ylab = "Probabilidade P(X=k)")

# --- 4. Sobrepor a Probabilidade Terica ---

# NO podemos usar 'curve(dpois(...))' porque a Poisson  DISCRETA.
# Em vez disso, calculamos a prob. P(X=k) para cada k e desenhamos PONTOS.

# Valores de k para calcular (de 0 at o mximo observado)
k_values <- 0:max(x)

# Probabilidades tericas P(X=k) para cada k
# Usamos dpois() para "density poisson"
prob_teoricas <- dpois(k_values, lambda = lambda0)

# Adiciona os PONTOS tericos ao grfico
points(k_values, prob_teoricas,
       col = "red",
       pch = 19,     # Crculo slido
       cex = 1.5)    # Tamanho do ponto

# (Opcional: Adiciona linhas verticais para criar um "lollipop chart")
segments(k_values, 0, k_values, prob_teoricas,
         col = "red",
         lwd = 2)

# --- 5. Adicionar 'rug' ---
# (Mostra os pontos de dados individuais na base)
# Usamos jitter() para que pontos sobrepostos (ex: muitos '3') fiquem visveis
rug(jitter(x))
