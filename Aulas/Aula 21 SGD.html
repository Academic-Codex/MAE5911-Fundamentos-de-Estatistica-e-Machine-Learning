# Modelo de rede neural para a mdia e para a varincia
model_mu_sigma2 <- torch::nn_module(
  initialize = function(in_dim, out_dim, N) {
    self$seq_mu <- torch::nn_sequential(
      torch::nn_linear(in_dim, round(N)),
      torch::nn_gelu(),
      torch::nn_linear(round(N), out_dim)
    )
    self$seq_sigma <- torch::nn_sequential(
      torch::nn_linear(in_dim, round(N)),
      torch::nn_gelu(),
      torch::nn_linear(round(N), out_dim),
    )
  },
  forward = function(x) {
    mu <- self$seq_mu(x)
    sigma2 <- torch_exp(self$logsigma2(x))
    return (list(mu=mu, sigma2))
  }
)

# Funo de estimao
library(IRdisplay)

estimacao <- function(x, y, model, epochs=1000, lr=0.1, f_true=NULL, print=TRUE) {
    n=length(y)
    yy = torch_tensor(y)$unsqueeze(-1)
    xx = torch_tensor(x)$unsqueeze(-1)

    optim = optim_adamw(model$parameters, lr=lr)
    loss_store = numeric()
    i = 0

    while(i < epochs) {
        i = i+1
        fit = model(xx)
        yhat = fit[[1]]
        sigma2 = fit[[2]]
        L = -(-1/2) * torch_sum(torch_log(sigma2) + ((yy - yhat)^2)/sigma2)
        optim$zero_grad()
        L$backward()
        optim$step()
        loss_store[i] = L$item()
        if(i%%50==0 & print) {
            clear_output(wait = TRUE)
            cat(paste0("Epoch: ", i, " Loss: ", as.numeric(L$item()), "\n"))
            par(mfrow=c(1,2))
            plot(y ~ x)
            points(as.matrix(yhat) ~x, col="tomato")
            if(!is.null(f_true)) {
                curve(f, add=TRUE, col="blue", lwd=2)
            }
            plot(loss_store, xlab="Epochs", ylab="Loss")
            Sys.sleep(0.1)
        }
    }
    invisible(NULL)
}
