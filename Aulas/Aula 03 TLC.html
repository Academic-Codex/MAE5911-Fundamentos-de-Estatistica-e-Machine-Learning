## --- parmetros ---
n      = 100      # tamanho da amostra em cada simulao
M      = 10000    # nmero de simulaes (experimentos)
theta0 = 1        # mdia verdadeira da Exponencial (pode trocar)

## Se quiser 0 aleatrio em cada execuo:
# theta0 <- runif(1, 1, 100)

## --- simulao ---
t = numeric(M)

for (i in 1:M) {
  z = rexp(n, rate = 1/theta0)                 # Exp(com mdia 0)
  t[i] = sqrt(n) * (mean(z) - theta0) / theta0 # padronizao  N(0,1)
}

## --- grfico ---
hist(t, prob = TRUE,
     main = "TCL: mdia da Exponencial (0 = mdia)",
     xlab = "t = sqrt(n) * (mean(Z) - 0) / 0")
