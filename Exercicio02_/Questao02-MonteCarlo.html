## Monte Carlo de cobertura para ICs (delta)  Exponencial(rate = theta)
## g_a(theta) = exp(-theta)
## g_b(theta) = exp(-0.1*theta) - exp(-theta)
## g_c(theta) = 1/theta^2

set.seed(123)

## ---- parmetros do experimento (ajuste  vontade) ----
theta0 <- 2.7027                 # use o que quiser; ex: o seu theta_MV dos dados
ns     <- c(6, 10, 20, 30, 50)   # tamanhos amostrais a testar
B      <- 10000                  # nmero de rplicas por (n, g)

## ---- funes g e derivadas ----
g_a      <- function(th)  exp(-th)
g_b      <- function(th)  exp(-0.1*th) - exp(-th)
g_c      <- function(th)  1/(th^2)

g_a_d    <- function(th) -exp(-th)                  # derivada de g_a
g_b_d    <- function(th) -0.1*exp(-0.1*th) + exp(-th)
g_c_d    <- function(th) -2/(th^3)

## varincia assinttica de theta_hat: Var(theta_hat) ~ theta^2 / n
## (Fisher info para Exponencial com taxa theta: I(theta)=n/theta^2)
var_theta_hat <- function(th, n) th^2 / n

## constri IC delta para g, com truncagem opcional ao espao natural
ci_delta <- function(g_hat, g_deriv_hat, var_th_hat,
                     alpha = 0.05, lower_trunc = -Inf, upper_trunc = Inf) {
  se2 <- (g_deriv_hat^2) * var_th_hat
  se  <- sqrt(se2)
  z   <- qnorm(1 - alpha/2)
  lo  <- g_hat - z * se
  hi  <- g_hat + z * se
  c(max(lo, lower_trunc), min(hi, upper_trunc))
}

## ---- motor de simulao para um n especfico ----
sim_cobertura_n <- function(n, theta, B) {
  cover_a <- cover_b <- cover_c <- 0L

  for (b in seq_len(B)) {
    z <- rexp(n, rate = theta)             # Z_i ~ Exp(theta)
    th_hat <- n / sum(z)                   # MLE: n / sum(Z_i)

    ## plug-in para g
    ghat_a <- g_a(th_hat)
    ghat_b <- g_b(th_hat)
    ghat_c <- g_c(th_hat)

    ## derivadas em th_hat
    g_a_d_hat <- g_a_d(th_hat)
    g_b_d_hat <- g_b_d(th_hat)
    g_c_d_hat <- g_c_d(th_hat)

    ## Var(theta_hat) ~ theta_hat^2 / n   (plug-in)
    var_th_hat <- var_theta_hat(th_hat, n)

    ## CIs (delta). Truncagem: [0,1] para probabilidades; [0, +inf) para varincia.
    ci_a <- ci_delta(ghat_a, g_a_d_hat, var_th_hat, lower_trunc = 0, upper_trunc = 1)
    ci_b <- ci_delta(ghat_b, g_b_d_hat, var_th_hat, lower_trunc = 0, upper_trunc = 1)
    ci_c <- ci_delta(ghat_c, g_c_d_hat, var_th_hat, lower_trunc = 0, upper_trunc = Inf)

    ## verdadeiros g(theta)
    ga_true <- g_a(theta)
    gb_true <- g_b(theta)
    gc_true <- g_c(theta)

    ## cobertura
    cover_a <- cover_a + as.integer(ga_true >= ci_a[1] && ga_true <= ci_a[2])
    cover_b <- cover_b + as.integer(gb_true >= ci_b[1] && gb_true <= ci_b[2])
    cover_c <- cover_c + as.integer(gc_true >= ci_c[1] && gc_true <= ci_c[2])
  }

  data.frame(
    n         = n,
    theta     = theta,
    cov_g_a   = cover_a / B,
    cov_g_b   = cover_b / B,
    cov_g_c   = cover_c / B,
    B         = B
  )
}

## ---- roda para todos os n ----
res <- do.call(rbind, lapply(ns, sim_cobertura_n, theta = theta0, B = B))
print(res, row.names = FALSE)

}

# -----------------------------
# Cobertura de ICs (mtodo delta)
# Modelo: Z ~ Exp(theta), theta > 0
# -----------------------------
set.seed(2025)

# Parmetros do experimento
theta_true = 2.7027               # voc pode mudar se quiser
ns         = c(6, 10, 30, 100, 1000) # tamanhos amostrais a testar
B          = 10000                # nmero de rplicas Monte Carlo
alpha      = 0.05
z_crit     = qnorm(1 - alpha/2)   # 1.96 para 95%

# Funes g() e derivadas g'()
g_a  <- function(th)  exp(-th)
gp_a <- function(th) -exp(-th)

g_b  <- function(th)  exp(-0.1*th) - exp(-th)
gp_b <- function(th) -0.1*exp(-0.1*th) + exp(-th)

g_c  <- function(th)  1/(th^2)
gp_c <- function(th) -2/(th^3)

# Construtor do IC delta para um dado g e g'
ci_delta <- function(theta_hat, n, g, gp) {
  gh  <- g(theta_hat)
  se  <- sqrt((theta_hat^2 / n) * (gp(theta_hat))^2)
  lo  <- gh - z_crit * se
  hi  <- gh + z_crit * se
  c(lo, hi)
}

# (opcional) truncar ao espao paramtrico do funcional
truncate_prob <- function(ci) pmin(1, pmax(0, ci))   # para probabilidades (0..1)
truncate_pos  <- function(ci) pmax(0, ci)           # para varincias (>=0)

# Verdadeiros valores dos funcionais em theta_true
g_true_a <- g_a(theta_true)
g_true_b <- g_b(theta_true)
g_true_c <- g_c(theta_true)

# Armazenar coberturas
cov_a <- cov_b <- cov_c <- numeric(length(ns))

# Loop nos tamanhos amostrais
for (k in seq_along(ns)) {
  n <- ns[k]

  hit_a <- hit_b <- hit_c <- logical(B)

  for (b in 1:B) {
    # 1) Gera amostra
    z  <- rexp(n, rate = theta_true)

    # 2) EMV da taxa: theta_hat = 1/mean(Z)
    theta_hat <- 1 / mean(z)

    # 3) ICs pelo delta
    ci_a <- ci_delta(theta_hat, n, g_a, gp_a)
    ci_b <- ci_delta(theta_hat, n, g_b, gp_b)
    ci_c <- ci_delta(theta_hat, n, g_c, gp_c)

    # 4) Truncar aos espaos naturais dos funcionais
    ci_a <- truncate_prob(ci_a)
    ci_b <- truncate_prob(ci_b)
    ci_c <- truncate_pos(ci_c)

    # 5) Verifica incluso do verdadeiro g(theta_true)
    hit_a[b] <- (g_true_a >= ci_a[1]) & (g_true_a <= ci_a[2])
    hit_b[b] <- (g_true_b >= ci_b[1]) & (g_true_b <= ci_b[2])
    hit_c[b] <- (g_true_c >= ci_c[1]) & (g_true_c <= ci_c[2])
  }

  # Cobertura emprica
  cov_a[k] <- mean(hit_a)
  cov_b[k] <- mean(hit_b)
  cov_c[k] <- mean(hit_c)
}

# Tabela resumida
tab <- data.frame(
  n      = ns,
  theta  = rep(theta_true, length(ns)),
  cov_g_a = round(cov_a, 4),
  cov_g_b = round(cov_b, 4),
  cov_g_c = round(cov_c, 4),
  B = B
)
print(tab)

#-----------------------------------------------------------------------
# Simulao de Monte Carlo para verificar a cobertura do IC para a 
# varincia de uma distribuio Exponencial.
#
# Disciplina: MAE5911 - Estatstica e ML
#-----------------------------------------------------------------------

# Parmetros da simulao
#------------------------------------------------
set.seed(123) # Para garantir que os resultados sejam reprodutveis

theta_verdadeiro = 2.0  # Valor real do parmetro
n = 1116                 # Tamanho amostral
N_replicas = 10000      # Nmero de simulaes de Monte Carlo
confianca = 0.95        # Nvel de confiana nominal do IC

# Clculo da quantidade de interesse verdadeira
#------------------------------------------------
g_theta_verdadeiro = 1 / theta_verdadeiro^2 # Var(Z) = 1/theta^2
cat("Valor verdadeiro da varincia (alvo):", g_theta_verdadeiro, "\n\n")

# Preparar a simulao
#------------------------------------------------
# Vetor para armazenar os resultados (1 se cobriu, 0 se no)
cobertura_vetor = numeric(N_replicas)
quantil_normal <- qnorm(1 - (1 - confianca) / 2) # Calcula o z_{alpha/2}, ~1.96
print(quantil_normal)

cat("Iniciando a simulao para n =", n, "...\n")

for (i in 1:N_replicas) {
  # 4a: Gerar uma amostra aleatria da populao Exp(theta)
  amostra_z <- rexp(n, rate = theta_verdadeiro)
  
  # 4b: Calcular a estimativa e o IC observado
  media_amostral <- mean(amostra_z)
  estimativa_g_theta <- media_amostral^2
  
  # Usamos a frmula do IC que derivamos: [g_hat * (1 +/- z * 2 / sqrt(n))]
  # Note que a varincia do estimador g_hat  4/theta^4, ento o desvio padro  2/theta^2.
  # A estimativa do desvio padro  2*g_hat = 2 * media_amostral^2
  margem_erro <- quantil_normal * (2 * estimativa_g_theta) / sqrt(n)
  
  limite_inferior <- estimativa_g_theta - margem_erro
  limite_superior <- estimativa_g_theta + margem_erro
  
  # 4c: Verificar se o IC cobriu o valor verdadeiro
  if (g_theta_verdadeiro >= limite_inferior && g_theta_verdadeiro <= limite_superior) {
    cobertura_vetor[i] <- 1
  } else {
    cobertura_vetor[i] <- 0
  }
}

cat("Simulao concluda.\n\n")

# Passo 5: Calcular e apresentar a cobertura emprica
#------------------------------------------------
cobertura_empirica <- mean(cobertura_vetor)

cat("Tamanho da amostra (n):", n, "\n")
cat("Nvel de confiana nominal:", confianca * 100, "%\n")
cat("Cobertura emprica estimada:", cobertura_empirica * 100, "%\n")
